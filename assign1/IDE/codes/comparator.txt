#include <Arduino.h>
bit A=B,A>B,A<B;
const int A0=2;
const int A1=3;
const int A2=4;
const int A3=5;
const int B0=6;
const int B1=7;
const int B2=8;
const int B3=9;
const int D=10;
const int C=11;
const int B=12;
const int A=13;
void setup()
{
    pinMode(A0,  INPUT);
    pinMode(A1,  INPUT);
    pinMode(A2,  INPUT);
    pinMode(A3,  INPUT);
    pinMode(B0,  INPUT);
    pinMode(B1,  INPUT);
    pinMode(B2,  INPUT);
    pinMode(B3,  INPUT);
    pinMode(D,  OUTPUT);
    pinMode(C,  OUTPUT);
    pinMode(B,  OUTPUT);
    pinMode(A,  OUTPUT);
}
void loop()
{
A==B =((!A0 ^ B0) && (!A1 ^ B1) && (!A2 ^ B2) && (!A3 ^ B3));
A>B  =((A3 && !B3)  (A2 && !B2  (!A3 ^ B3)  (A1 && !B1) && (!A3 ^ B3) && (!A2 ^ B2)  (A0 && !B0) && (!A3 ^ B3) && (!A2 ^ B2) && (!A1 ^ B1)); 
A<B  =(!A3 && B3)  ((!A2 && B2)&&(!(A3 ^ B3)))  (!A1 && B1) && ((!(A3 ^ B3)) && (!(A2 ^ B2))) || (!A0 && B0) && ((!(A3 ^B3)) && (!(A2 ^ B2)) && (!(A1 ^B1)))
if(A=B)
D=high;C=high;A=low;B=low;
if(A<B)
D=high;C=low;A=low;B=low;
if(A<B)
D=low;C=high;A=low;B=low;
}
